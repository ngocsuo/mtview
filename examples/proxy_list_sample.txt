# Sample Proxy List
# Format: protocol://host:port hoặc protocol://host:port:username:password
# Hỗ trợ: socks5, http, https

# SOCKS5 proxies không auth
socks5://127.0.0.1:1080
socks5://192.168.1.100:1080
socks5://10.0.0.50:9050

# SOCKS5 proxies có auth
socks5://proxy1.example.com:1080:user1:pass1
socks5://proxy2.example.com:1080:user2:pass2

# HTTP proxies không auth
http://127.0.0.1:8080
http://192.168.1.101:3128

# HTTP proxies có auth
http://proxy3.example.com:8080:user3:pass3
http://proxy4.example.com:3128:user4:pass4

# HTTPS proxies (sẽ được chuyển thành HTTP)
https://proxy5.example.com:443:user5:pass5
